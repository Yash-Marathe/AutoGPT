import time
from IPython.display import clear_output, HTML, display, Color
def print_spin(text, delay=0.1):
    animation = ["|", "/", "-", "\\"]
    for char in text:
        print(char, end="\r")
        time.sleep(delay)
    for i in range(100):
        print(animation[i % 4], end="\r")
        time.sleep(delay)
    print("")

def print_color(text, color="green"):
    print(Color.text(text, color))

print_spin("Hello World! ", delay=0.05)
print_color("Hello World! ", color="red")
